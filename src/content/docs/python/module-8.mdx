1. ကွန်ပျူတာက 1 ကနေ 100 ကြား ဂဏန်းတစ်ခုကို ကျပန်း ရွေးထားပါမည်
2. ကစားသမားက မှန်းပြီး ရိုက်ထည့်ရပါမည်

4. အဖြေမှန်ရင် ဘယ်နှစ်ကြိမ်မြောက်မှာ မှန်သလဲ ပြပါမည်

### Version 1 — အခြေခံ Game

```python
import random

def play_game():
    print("=====================================")
    print("ဂဏန်းခန့်မှန်း ဂိမ်းမှ ကြိုဆိုပါတယ်!")
    print("ကွန်ပျူတာ 1 ကနေ 100 ကြား ဂဏန်းတစ်ခုကို ကျပန်း ရွေးထားပါတယ်")
    print("=====================================\n")

    secret_number = random.randint(1, 100)  # Line 9: ကျပန်း ဂဏန်းရွေး
    attempts = 0

    while True:                              # Line 12: အဖြေမမှန်မချင်း Loop ဆက်
        try:
            guess = int(input("သင် ခန့်မှန်းမည့် ဂဏန်း: "))  # Line 14: Input ယူ
            attempts += 1                    # Line 15: ကြိမ်ရေ တိုး

            if guess < secret_number:        # Line 17: ငယ်နေသလား စစ်ဆေး
                print("📉 နည်းနည်း ငယ်နေသေးတယ်! ထပ်ကြိုးစားကြည့်ပါ\n")
            elif guess > secret_number:      # Line 19: ကြီးနေသလား စစ်ဆေး
                print("📈 နည်းနည်း ကြီးနေတယ်! ထပ်ကြိုးစားကြည့်ပါ\n")
            else:                            # Line 21: မှန်သည် (တူညီသည်)
                print(f"\n🎉 မှန်ပါတယ်! ဂဏန်း {secret_number} ဖြစ်ပါတယ်")
                print(f"သင် {attempts} ကြိမ်မြောက်မှာ မှန်ကြောင်း ရှာတွေ့ပါသည်!")
                break                        # Line 24: Loop ရပ်

        except ValueError:                   # Line 26: ဂဏန်းမဟုတ်ရင်
            print("❌ ကျေးဇူးပြု၍ ဂဏန်းသာ ရိုက်ထည့်ပါ\n")

play_game()
```

### Version 2 — ပိုကောင်းသော Game (Features ထည့်ပြီး)

```python
import random

def get_difficulty():
    """Difficulty Level ရွေးချယ်ခြင်း"""
    print("Difficulty ရွေးပါ:")
    print("  [1] လွယ်   — 1-50,  Chances 10")
    print("  [2] ပုံမှန် — 1-100, Chances 7")
    print("  [3] ခက်   — 1-200, Chances 5")
    
    while True:
        choice = input("ရွေးချယ်ပါ (1/2/3): ").strip()
        if choice == "1":
            return 50, 10, "လွယ်"
        elif choice == "2":
            return 100, 7, "ပုံမှန်"
        elif choice == "3":
            return 200, 5, "ခက်"
        else:
            print("1, 2, 3 ကိုသာ ရွေးပါ")

def evaluate_performance(attempts, max_attempts):
    """ကစားသူ ကိုဆန်းစစ်ပြီး Rating ပေးခြင်း"""
    ratio = attempts / max_attempts
    if ratio <= 0.3:
        return "⭐⭐⭐ Excellent!"
    elif ratio <= 0.6:
        return "⭐⭐ Good!"
    else:
        return "⭐ You can do better!"

def play_game():
    print("=" * 45)
    print("   ဂဏန်းခန့်မှန်း ဂိမ်းမှ ကြိုဆိုပါတယ်!")
    print("=" * 45 + "\n")
    
    max_number, max_attempts, level = get_difficulty()
    secret_number = random.randint(1, max_number)
    attempts = 0
    history = []   # Guess History သိမ်းခြင်း
    
    print(f"\n{level} Level — 1 မှ {max_number} ကြား ဂဏန်း ({max_attempts} ကြိမ် Chance)")
    
    while attempts < max_attempts:
        remaining = max_attempts - attempts
        print(f"\nကျန် Chance: {remaining}")
        
        try:
            guess = int(input("ဂဏန်း ခန့်မှန်းပါ: "))
            
            if not 1 <= guess <= max_number:
                print(f"❌ 1 မှ {max_number} ကြား ဂဏန်းသာ ထည့်ပါ")
                continue
            
            attempts += 1
            history.append(guess)
            
            if guess < secret_number:
                diff = secret_number - guess
                hint = "အနည်းငယ် ငယ်" if diff <= 10 else "အများကြီး ငယ်"
                print(f"📉 {hint}နေသေးတယ်!")
            elif guess > secret_number:
                diff = guess - secret_number
                hint = "အနည်းငယ် ကြီး" if diff <= 10 else "အများကြီး ကြီး"
                print(f"📈 {hint}နေသေးတယ်!")
            else:
                rating = evaluate_performance(attempts, max_attempts)
                print(f"\n{'=' * 45}")
                print(f"🎉 မှန်ပါတယ်! ဂဏန်း {secret_number} ဖြစ်ပါတယ်")
                print(f"ကြိမ်အရေအတွက်: {attempts} / {max_attempts}")
                print(f"Rating: {rating}")
                print(f"Guess History: {history}")
                print(f"{'=' * 45}")
                return True
        
        except ValueError:
            print("❌ ဂဏန်းသာ ရိုက်ပါ")
    
    print(f"\n😢 Chances ကုန်သွားပါပြီ! ဂဏန်းဟာ {secret_number} ဖြစ်ပါတယ်")
    return False

def main():
    while True:
        won = play_game()
        
        print("\nနောက်ထပ် ကစားမည်လား?")
        again = input("(yes/no): ").strip().lower()
        if again not in ("yes", "y", "ဟုတ်"):
            print("\n👋 ကစားပေးသည့်အတွက် ကျေးဇူးတင်ပါသည်!")
            break

main()
```
