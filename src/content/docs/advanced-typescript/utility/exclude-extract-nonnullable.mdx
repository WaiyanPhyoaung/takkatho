let resolved: ResolvedStatus = "error"; // ဒါက အလုပ်လုပ်တယ်

};`

export const nonnullable =`let currentStatus: Status = "pending";

// ... တချို့ Logic တွေ ...

if (currentStatus !== null && currentStatus !== undefined) {
// ဒီ Block ထဲမှာဆိုရင် TypeScript က currentStatus ဟာ null ဒါမှမဟုတ် undefined မဟုတ်တော့ဘူးဆိုတာ သိနေပြီ။
// တခြား Function တစ်ခုကို ပို့ရင်လည်း Type ကို အတိအကျ သတ်မှတ်ပေးလို့ ရတယ်:
let activeStatus: NonNullable<Status> = currentStatus;
console.log(activeStatus.toUpperCase()); // toUpperCase() ကို စိတ်ချစွာ သုံးလို့ရပြီ။
};`

<Aside title='ရည်ရွယ်ချက်' type="tip">
  `Exclude<T, U>`, `Extract<T, U>`, နဲ့ `NonNullable<T>` Utility Types တွေကို ဘယ်လို သုံးရသလဲဆိုတာ နားလည်ဖို့ ဖြစ်ပါတယ်။ သူတို့က Union Types တွေနဲ့ အလုပ်လုပ်တာပါ။ `Pick<T, K>` နဲ့ `Omit<T, K>` တို့ကတော့ Object Types တွေရဲ့ Properties (Keys) တွေနဲ့ အလုပ်လုပ်တာပါ။ သဘောတရားကတော့ ဆင်တူပါတယ်။
</Aside>

ဒီဥပမာတွေအတွက် Union Type လေးတစ်ခုကို အရင်သတ်မှတ်ထားရအောင်။

## `Exclude<T, U>`: Union ထဲက Types တွေကို ဖယ်ရှားခြင်း

- **ဘာလုပ်ပေးသလဲ**: `Exclude<T, U>` က `T` ဆိုတဲ့ Union Type ထဲက `U` နဲ့ တူညီတဲ့ Types တွေကို ဖယ်ထုတ်ပြီး Type အသစ်တစ်ခု လုပ်ပေးပါတယ်။
- **ဘယ်လို သုံးမလဲ**: `Exclude<Status, "processing" | "pending">`
- **ရလာမယ့် Type**: `"success" | "error" | null | undefined`
- **ဘယ်အခါသုံးမလဲ**: အခြေအနေ (Status) တွေ အများကြီး ရှိနေတယ်။ ဒါပေမဲ့ တချို့ အခြေအနေတွေကိုပဲ ဂရုစိုက်ချင်ပြီး ကျန်တာတွေကို လျစ်လျူရှုချင်တဲ့အခါမျိုးမှာ သုံးနိုင်ပါတယ်။

<Code lang="typescript" code={exclude} title="index.ts" />

## `Extract<T, U>`: Union ထဲက Types တွေကို ရွေးချယ်ခြင်း

- **ဘာလုပ်ပေးသလဲ**: `Extract<T, U>` က `T` ဆိုတဲ့ Union Type ထဲက `U` နဲ့ တူညီတဲ့ Types တွေကို ရွေးချယ်ပြီး Type အသစ်တစ်ခု လုပ်ပေးပါတယ်။
- **ဘယ်လို သုံးမလဲ**: `Extract<Status, "success" | "error">`
- **ရလာမယ့် Type**: `"success" | "error"`
- **ဘယ်အခါသုံးမလဲ**: ကျယ်ပြန့်တဲ့ Union Type တစ်ခု ရှိနေတယ်။ ဒါပေမဲ့ အဲဒီထဲက အစိတ်အပိုင်း တချို့ကိုပဲ သီးသန့် လိုချင်တဲ့အခါမျိုးမှာ သုံးနိုင်ပါတယ်။

## `NonNullable<T>`: `null` နှင့် `undefined` ကို ဖယ်ရှားခြင်း

- **ဘာလုပ်ပေးသလဲ**: `NonNullable<T>` က `T` ဆိုတဲ့ Type ထဲက `null` နဲ့ `undefined` ကို ဖယ်ရှားပြီး Type အသစ်တစ်ခု လုပ်ပေးပါတယ်။
- **ဘယ်လို သုံးမလဲ**: `NonNullable<Status>`
- **ရလာမယ့် Type**: `"pending" | "processing" | "success" | "error"`
- **ဘယ်အခါသုံးမလဲ**: တန်ဖိုးတစ်ခုက `null` ဒါမှမဟုတ် `undefined` ဖြစ်နိုင်တယ်။ ဒါပေမဲ့ Code ထဲမှာ စစ်ဆေးပြီးနောက်ပိုင်း `null` ဒါမှမဟုတ် `undefined` မဟုတ်တော့ဘူးလို့ သေချာတဲ့အခါမျိုးမှာ အရမ်း အသုံးဝင်ပါတယ်။
