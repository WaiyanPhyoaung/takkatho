export const code = `
// number Array အတွက် ပထမဆုံး Element ယူတဲ့ Function
function getFirstNumber(arr: number[]): number | undefined {
  return arr[0];
}

// string Array အတွက် ပထမဆုံး Element ယူတဲ့ Function
function getFirstString(arr: string[]): string | undefined {
return arr[0];
}

// Object Array အတွက် ပထမဆုံး Element ယူတဲ့ Function
interface User { name: string; age: number; }

function getFirstUser(arr: User[]): User | undefined {
return arr[0];
}

// အခု ဒီ Function သုံးခုလုံးက ပထမဆုံး Element ကိုပဲ ယူတဲ့ အလုပ်ကိုပဲ လုပ်တာပါ။
`

အခြေခံ TypeScript က Error အများကြီးကို ကာကွယ်ပေးနိုင်ပေမဲ့ Program တွေ၊ Project တွေ ပိုကြီးလာ၊ ပိုရှုပ်ထွေးလာတဲ့အခါ အခက်အခဲအသစ်တွေ ပေါ်လာတတ်ပြန်ပါတယ်။

- **Code တွေ ထပ်ရေးနေရခြင်း (Repeating)**: ပုံစံတူ အလုပ်တစ်ခုတည်းကိုပဲ Data အမျိုးမျိုး (`number`, `string`, `Object`) အတွက် သီးသန့် Code တွေ ခွဲရေးနေရတာမျိုးပါ။ ဒါက Code တွေ အများကြီး ဖြစ်လာပြီး ပြင်ရခက်စေပါတယ်။
- **Type တွေ အများကြီး ထပ်လုပ်နေရခြင်း**: ရှိပြီးသား Data ပုံစံ (Type) ကနေ နည်းနည်းလေး ပြောင်းသုံးချင်တာမျိုး (ဥပမာ - Property တချို့လောက်ပဲ လိုတာမျိုး) ဆိုရင် Data ပုံစံအသစ် (Type အသစ်) တွေ ထပ်ခွဲရေးနေရတာမျိုး ဖြစ်ပါတယ်။
- **Data အသေးစိတ်ကို မပျောက်စေဘဲ စစ်ဆေးခြင်း**: Data တစ်ခုက ကိုယ်လိုချင်တဲ့ ပုံစံနဲ့ ကိုက်ညီရဲ့လား စစ်ချင်တယ်။ ဒါပေမဲ့ အဲဒီ Data ရဲ့ တိကျတဲ့ တန်ဖိုးလေးတွေကိုလည်း မပျောက်စေချင်ဘူး (ဥပမာ - `number` Type ဆိုတာထက် Constant တန်ဖိုး `"5"` ဆိုတာကိုပဲ သုံးချင်တာမျိုး) ပါ။

<Aside title="ရှင်းပြချက်">
  မြင်တဲ့အတိုင်းပဲ Function သုံးခုလုံးက `return arr[0];` ဆိုတဲ့ အလုပ်
  တစ်ခုတည်းကိုပဲ လုပ်တာပါ။ ဒါပေမဲ့ Array ထဲက Data Type မတူတာကြောင့် Function
  သုံးခု ခွဲရေးလိုက်ရတယ်။ ဒါက Code Duplication (Code ထပ်ခြင်း) ပါပဲ ခင်ဗျာ။
</Aside>

ဒီလို Code တွေ ထပ်နေတော့ ဘာဖြစ်လဲဆိုတော့ - အကယ်၍ ဒီ Function ရဲ့ အလုပ်လုပ်ပုံလေး နည်းနည်း ပြောင်းချင်တယ် ဆိုပါစို့ (ဥပမာ - Array က `null` ဖြစ်နေလား အရင်စစ်ချင်တာမျိုး)။ အဲဒီအခါကျရင် ကျွန်တော်တို့ ခုနက ရေးထားတဲ့ `getFirstNumber`, `getFirstString`, `getFirstUser` ဆိုတဲ့ Function သုံးခုလုံးကို လိုက်ပြင်နေရတော့မှာပါ။ Code တွေ များလာလေ၊ ပြင်ရတာ ပိုရှုပ်ပြီး ပိုမှားနိုင်လေပါပဲ ခင်ဗျာ။

ဒီအခက်အခဲတွေက Code ရေးရတာ ပိုများစေတယ်၊ နားလည်ရ ခက်စေတယ်၊ ပြင်ရတာလည်း ပိုကြာစေပါတယ်။ ဒီလိုဆိုရင် TypeScript သုံးရခြင်းရဲ့ အကျိုးကျေးဇူးလည်း အပြည့်အဝ မရနိုင်တော့ပါဘူး။
