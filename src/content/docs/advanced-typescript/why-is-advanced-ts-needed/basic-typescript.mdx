export const code = `
interface User {
  name: string;
  age: number;
}

function greetUser(user: User): void {
  // အခု user က User Type ဖြစ်မှန်း၊ name က string ဖြစ်မှန်း TypeScript က တိတိကျကျ သိနေပြီ
  console.log(\`Hello, \${user.name.toUpperCase()}!\`);
}

let goodUserData: User = { name: "Alice", age: 30 }; // မှန်ကန်တယ်

// let badUserData: User = { name: "Bob" }; // ERROR: age မပါဘူးလို့ TypeScript က ကြိုပြောပြီ!
// greetUser(null); // ERROR: null က User Type မဟုတ်ဘူးလို့ ကြိုပြောပြီ!

greetUser(goodUserData); // ပြဿနာမရှိ အလုပ်လုပ်တယ်
`

ဒီလိုမျိုး အစကတည်းက မသိဘဲ Run တဲ့အချိန်ကျမှ Error တက်တာတွေကို ကာကွယ်ဖို့အတွက် TypeScript က အများကြီး ကူညီပေးနိုင်ပါတယ်။ TypeScript က ကျွန်တော်တို့ကို Code မရေးခင်မှာ ကိုယ့်ရဲ့ Data တွေက ဘယ်လို ပုံစံမျိုး၊ ဘယ်လို Type မျိုး ဖြစ်ရမယ် ဆိုတာကို သေချာ ကြိုတင် သတ်မှတ်ခိုင်းပါတယ်။

ဥပမာ - `User` ဆိုတဲ့ Data ဆိုရင် သူ့မှာ `name` ဆိုတဲ့ စာသား (String) နဲ့ `age` ဆိုတဲ့ ဂဏန်း (Number) မဖြစ်မနေ ပါဝင်ရမယ် စသဖြင့် သတ်မှတ်တာပေါ့။ ဒါကို **Static Typing** လို့ ခေါ်ပါတယ်။

ဒီလို သတ်မှတ်လိုက်တော့ ဘာကောင်းသွားလဲ ဆိုတော့ - Code ရေးနေတုန်းမှာတင် (Editor ထဲမှာတင်) ဒါမှမဟုတ် Compile လုပ်နေတုန်းမှာတင် Type မှားယွင်းနေတာတွေ၊ လိုအပ်တဲ့ Property တွေ မပါတာတွေကို TypeScript က ကြိုတင် သတိပေး (Error ပြ) နိုင်သွားတာပါပဲ။

ဒီ Code လေးကို TypeScript နဲ့ ရေးကြည့်ရအောင် -

### 💡 ရှင်းပြချက်

TypeScript ဟာ ကျွန်တော်တို့ ကြိုတင် သတ်မှတ်ထားတဲ့ Type စည်းမျဉ်းတွေ အတိုင်း ကိုက်ညီမှု ရှိမရှိကို Code ရေးနေတုန်းကတည်းက စစ်ဆေးပေးပါတယ်။ ဒါကြောင့် Program ကြီး Run တဲ့အချိန်ကျမှ Error ဖြစ်ပြီး ပျက်ကျသွားမယ့် အစား၊ အစောကြီးကတည်းက သိပြီး ကြိုတင် ပြင်ဆင်လို့ ရသွားတာပေါ့။ ဒါက Application တွေ ရေးရတာကို ပိုပြီး ခိုင်မာ စိတ်ချရစေပါတယ်။
