export const code = `
function getFirst<T>(arr: T[]): T | undefined {
  return arr[0];
}

// အခုဆို ဒီ getFirst Function လေးကို Type အမျိုးမျိုးနဲ့ သုံးလို့ရပြီ

let firstNumber = getFirst([1, 2, 3]); // number Array နဲ့ သုံးလို့ရပြီ
// firstNumber ရဲ့ Type က number | undefined လို့ TypeScript က သိတယ်

let firstString = getFirst(["a", "b", "c"]); // string Array နဲ့လည်း သုံးလို့ရပြီ
// firstString ရဲ့ Type က string | undefined လို့ TypeScript က သိတယ်

interface User { name: string; age: number; }
let users: User[] = [{ name: "Alice", age: 30 }, { name: "Bob", age: 25 }];
let firstUser = getFirst(users); // ကိုယ်ပိုင် Object Array နဲ့လည်း သုံးလို့ရပြီ
// firstUser ရဲ့ Type က User | undefined လို့ TypeScript က သိတယ်
`

ဒီလို အခက်အခဲတွေကို ဖြေရှင်းဖို့ TypeScript မှာ ပိုအစွမ်းထက်တဲ့ Features တွေ ရှိပါတယ်။ အဲဒါတွေကတော့ Generics, Utility Types နဲ့ `satisfies` Keyword တွေပဲ ဖြစ်ပါတယ်။

### Generics

Code တစ်ခုတည်းကိုပဲ Data Type အမျိုးမျိုးနဲ့ ပြန်လည် အသုံးပြုနိုင်အောင် လုပ်ပေးပါတယ်။ ဒီတော့ Code တွေ ထပ်ရေးရတာ သက်သာသွားတာပေါ့။ (ပုံစံတူ လုပ်ဆောင်ချက်ကို မတူညီတဲ့ Type တွေနဲ့ သုံးနိုင်စေတဲ့ မှော်ဆန်တဲ့ Box လိုမျိုးပေါ့ ခင်ဗျာ။)

### Utility Types

ရှိပြီးသား Data ပုံစံ (Type) တွေကနေ လိုအပ်သလို ပြောင်းလဲထားတဲ့ Type အသစ်တွေကို အလွယ်တကူ မြန်မြန် ဖန်တီးနိုင်စေပါတယ်။ (Data ပုံစံတွေကို အမြန် ပြင်ဆင်နိုင်တဲ့ Tools မျိုးစုံ ပါတဲ့ Box လိုမျိုးပါပဲ။)

<Aside title="မှတ်ချက်">
  ဒီအရာတွေကို လေ့လာခြင်းအားဖြင့် Program တွေ ရေးတဲ့အခါ Code တွေ ပိုပြီး
  သပ်ရပ်လာမယ်၊ ပြန်သုံးလို့ရတာ (Reusable) ပိုများလာမယ်၊ ထိန်းသိမ်းရတာ (Maintainable)
  ပိုလွယ်ကူလာပါလိမ့်မယ်။ အရေးကြီးဆုံးကတော့ ပိုပြီး ခိုင်မာစိတ်ချရတဲ့
  Program တွေ ရေးနိုင်လာမှာ ဖြစ်ပါတယ် ခင်ဗျာ။
</Aside>

ဥပမာ - အရင် သင်ခန်းစာက Code တွေ ထပ်နေတဲ့ ပြဿနာ (Type အမျိုးမျိုးအတွက် Array ထဲက ပထမဆုံး Element ယူတဲ့ Function တွေကို ခွဲရေးရတာ) ကို မှတ်မိသေးတယ် မဟုတ်လား ခင်ဗျာ?

အဲဒီလို ပြဿနာမျိုးအတွက် Generics ဆိုတာကို သုံးလို့ ရပါတယ်။ Generics နဲ့ဆိုရင် အဲဒီ Function ကို Code တစ်ကြောင်းတည်းနဲ့တင် Type အမျိုးမျိုးအတွက် လွယ်လွယ်လေး ရေးလို့ရသွားပါပြီ။ အောက်က Code လေးကို ကြည့်လိုက်ပါ ခင်ဗျာ -
