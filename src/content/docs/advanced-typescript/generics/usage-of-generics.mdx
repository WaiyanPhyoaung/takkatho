// Generic Type Variable <T> ကို သုံးမယ်
function wrapInArray<T>(input: T): T[] {
  return [input];
}

// wrapInArray Function ကို သုံးကြည့်မယ်
let numberArray = wrapInArray(10); // 10 ဆိုတဲ့ number ကို ထည့်လိုက်တော့ T က number လို့ မှန်းသိတယ်၊ ပြန်ထွက်တဲ့ Type က number[] ဖြစ်သွားတယ်
console.log(numberArray); // Output: [10]

let stringArray = wrapInArray("World"); // "World" ဆိုတဲ့ string ကို ထည့်လိုက်တော့ T က string ဖြစ်သွားတယ်၊ ပြန်ထွက်တဲ့ Type က string[] ဖြစ်သွားတယ်
console.log(stringArray); // Output: ["World"]

let userObject = { id: 1, name: "Admin" };
let userArray = wrapInArray(userObject); // { id: number; name: string; } ဆိုတဲ့ Object ထည့်လိုက်တော့ T က အဲ့ဒီ Object Type ဖြစ်သွားတယ်၊ ပြန်ထွက်တဲ့ Type က အဲ့ဒီ Object Type ရဲ့ array[] ဖြစ်သွားတယ်
console.log(userArray); // Output: [{ id: 1, name: "Admin" }]

export const classGenerics = `
// Stack Class ကို Generic ဖြစ်အောင် <ElementType> သုံးလိုက်မယ်
// ဒီ Stack ထဲမှာ ဘယ်လို Type မျိုးနဲ့ Data တွေ ထည့်မလဲဆိုတာ ElementType က သတ်မှတ်ပေးမယ်
class Stack<ElementType> {
  // ဒီ Stack ထဲက data တွေကို သိမ်းဆည်းပေးမယ့် array
  private items: ElementType[] = [];

  // Stack ထဲကို Data ထည့်မယ့် Function
  push(item: ElementType): void { // ထည့်မယ့် item ရဲ့ Type ဟာ Stack ရဲ့ ElementType အတိုင်း ဖြစ်ရမယ်
    this.items.push(item);
  }

  // Stack ထဲက နောက်ဆုံး Data ကို ပြန်ထုတ်မယ့် Function
  pop(): ElementType | undefined { // ပြန်ထွက်လာမယ့် Data ရဲ့ Type က Stack ရဲ့ ElementType ပဲ ဖြစ်မယ်၊ ဒါမှမဟုတ် ဘာမှ မရှိရင် undefined ပေါ့
    return this.items.pop();
  }

  // Stack ထဲက နောက်ဆုံး Data ကို ကြည့်မယ့် Function (အပြင်ကို ထုတ်ယူခြင်း မရှိ)
  peek(): ElementType | undefined { // Type က pop လိုပဲ
    return this.items[this.items.length - 1];
  }
}

// အခု ဒီ Generic Stack Class ကို အသုံးပြုပြီး Stack Object တွေ ဖန်တီးကြည့်မယ်

// number တွေပဲ ထည့်လို့ရမယ့် Stack တစ်ခု ဖန်တီးမယ်
let numberStack = new Stack<number>(); // Stack<number> လို့ သတ်မှတ်လိုက်တော့ ElementType ဟာ number လုံးလုံး ဖြစ်သွားပြီ

numberStack.push(10); // number ထည့်လို့ရတယ်
numberStack.push(20); // number ထပ်ထည့်လို့ရတယ်

let lastNumber = numberStack.pop(); // pop လုပ်လိုက်ရင် number | undefined ပြန်ထွက်မယ်။ တကယ်တော့ 20 (number) ပြန်ထွက်လာတယ်
console.log(lastNumber); // Output: 20

// string တွေပဲ ထည့်လို့ရမယ့် Stack တစ်ခု ဖန်တီးမယ်
let stringStack = new Stack<string>(); // Stack<string> လို့ သတ်မှတ်လိုက်တော့ ElementType က string ဖြစ်သွားပြီ

stringStack.push("first"); // string ထည့်လို့ရတယ်
stringStack.push("second"); // string ထပ်ထည့်လို့ရတယ်

let lastString = stringStack.pop(); // pop လုပ်လိုက်ရင် string | undefined ပြန်ထွက်မယ်။ တကယ်တော့ "second" (string) ပြန်ထွက်လာတယ်
console.log(lastString); // Output: "second"

// stringStack.push(123); // ERROR: stringStack က string တွေအတွက် သီးသန့် Stack ဖြစ်လို့ number သွားထည့်လို့ မရတော့ဘူး။ TypeScript က ချက်ချင်း Error ပြပါမယ်!
`

## Generic Classes

Classes တွေမှာလည်း Generics ကို သုံးလို့ ရပါတယ်။ ဥပမာအားဖြင့် Data တွေကို အစီအစဉ်အတိုင်း ထည့်သွင်းပြီး၊ နောက်ဆုံးမှ ထည့်တဲ့ Data ကို အရင်ဆုံး ပြန်ထုတ်ယူရတဲ့ Data Structure တစ်ခုဖြစ်တဲ့ Stack (Last-In, First-Out) ရိုးရှင်းတဲ့ ပုံစံတစ်ခုကို Class နဲ့ တည်ဆောက်ကြည့်ရအောင်。
