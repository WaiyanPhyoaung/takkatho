// အရင်က ပြောခဲ့တဲ့ ApiResponse Interface (Data ရဲ့ Type ကို Generic လုပ်ထားတာ)
interface ApiResponse<DataType> {
  data: DataType; 
  success: boolean; 
  error?: string; 
}

// API ကနေ User Data ပြန်လာရင် ဘယ်လိုပုံစံ ဖြစ်မယ်ဆိုတာ သတ်မှတ်ချက်
interface User {
  id: number;
  name: string;
  email: string;
}

// API ကနေ Data ယူတဲ့ Generic Function (နားလည်ရလွယ်အောင် ရိုးရှင်းအောင် ရေးထားတာပါ)
// ဒီ Function က ResponseType ဆိုတဲ့ Generic Type နဲ့ အလုပ်လုပ်မယ်
// ပြန်ထွက်လာမယ့် Data ကတော့ ApiResponse<ResponseType> ပုံစံ ဖြစ်မယ်လို့ Promise (ကတိ) ပေးထားတယ်
async function fetchData<ResponseType>(url: string): Promise<ApiResponse<ResponseType>> {
  try {
    const response = await fetch(url);
    if (!response.ok) {
      throw new Error(\`HTTP error! status: \${response.status}\`);
    }
    const data: ApiResponse<ResponseType> = await response.json();
    if (!data.success) {
      throw new Error(data.error || "API returned success: false");
    }
    return data;
  } catch (error) {
    const message = error instanceof Error ? error.message : "Unknown fetch error";
    // data ကို null လို့ သတ်မှတ်ထားတယ်၊ ဒါပေမယ့် ResponseType အနေနဲ့ ယူဆမယ်လို့ 'as ResponseType' နဲ့ ပြောထားတယ်
    return { data: null as ResponseType, success: false, error: message };
  }
}

// ဒီ Generic fetchData Function ကို လက်တွေ့ ဘယ်လို သုံးလဲ ကြည့်ရအောင်
async function getUser(userId: number) {
  // fetchData ကို ခေါ်တဲ့အခါ ApiResponse ထဲက data ဟာ User Type ဖြစ်ရမယ်လို့ <User> ဆိုပြီး ပြောလိုက်တယ်
  const userResponse = await fetchData<User>(\`/api/users/\${userId}\`);
  if (userResponse.success) {
    // userResponse.data ရဲ့ Type ဟာ User ဖြစ်မှန်း TypeScript က အတိအကျ သိနေပြီ။ ဒါ့ကြောင့် .name ကို ရဲရဲဝံ့ဝံ့ သုံးလို့ရတယ်
    console.log("Fetched User:", userResponse.data.name);
  } else {
    // Error ရှိရင်လည်း userResponse.error ကို သုံးလို့ရတယ်
    console.error("Failed to fetch user:", userResponse.error);
  }
}
`;

### Data များကို Generic ပုံစံဖြင့် ဆွဲယူခြင်း (Fetching Data)

ကျွန်တော်တို့ Application တွေ ရေးတဲ့အခါ အပြင်က API (Application Programming Interface) တွေဆီကနေ Data တွေ သွားယူရတာကို အမြဲလိုလို လုပ်ရပါတယ်။ ဒီ Data ယူတဲ့ Logic (အလုပ်လုပ်ပုံ) တစ်ခုလုံးကို Generic ဖြစ်အောင် ရေးထားတာက အရမ်းကို အသုံးဝင်ပြီး၊ လုပ်ငန်းခွင်မှာ လုပ်ရိုးလုပ်စဉ် (Best Practice) လိုတောင် ဖြစ်နေပါပြီ။

အရင် သင်ခန်းစာမှာ ကျွန်တော်တို့ `ApiResponse` ဆိုတဲ့ Generic Interface လေးတစ်ခု သတ်မှတ်ခဲ့တာ မှတ်မိဦးမယ် ထင်ပါတယ်။ API ကနေ ဘယ်လို Data အမျိုးအစားပဲ ပြန်လာလာ၊ ဒီ `ApiResponse` ပုံစံအတိုင်း အလွယ်တကူ ထည့်သုံးလို့ ရတာပေါ့။

### UI Components များ (ဥပမာ - React)

Website ဒါမှမဟုတ် Application တွေအတွက် UI (User Interface) Components တွေ ရေးတဲ့အခါ (ဥပမာ - Lists တွေ၊ Tables တွေ၊ Dropdowns တွေ) Data အမျိုးမျိုးကို ပြသနိုင်တဲ့ Components တွေ လိုအပ်လာတတ်ပါတယ်။ ဒီလို Components တွေကို Generic သုံးပြီး ရေးတာက တကယ်ကို လက်တွေ့ကျကျ အသုံးဝင်ပါတယ်။
